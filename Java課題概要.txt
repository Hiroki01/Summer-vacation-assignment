・テーブル１（生徒用）
CREATE TABLE Student (
		id INT(9) NOT NULL,
		name VARCHAR(64) NOT NULL,
		gender VARCHAR(5) NOT NULL,
		email VARCHAR(64) NOT NULL,
		department VARCHAR(20) NOT NULL,
		course VARCHAR(30) NOT NULL,
		qualification_id INT(5),
		pass VARCHAR(64) NOT NULL,
		PRIMARY KEY (id));
